    int cnt = read(0, buf, len);

    if (buf[cnt - 1] == '\n')
        buf[cnt - 1] = '\0';

void add_to_cart()
{

    int id = read_int();

    {
        puts("Invalid product ID!");
        return;
    }

    printf("Quantity: ");
    int quantity = read_int();

    if (quantity <= 0)
    {
        puts("Invalid quantity!");
        return;
    }

    cart_item *new_item = (cart_item *)malloc(sizeof(cart_item));
    if (!new_item)
    {
        puts("Malloc failed!");
        return;
    }

    new_item->item = products[id];
    new_item->quantity = quantity;
    new_item->next = cart_head;
    cart_head = new_item;
    printf("Added %d %s to cart\n", quantity, products[id]->name);
}

    while (curr != NULL)
    {
        total += curr->item->price * curr->quantity;
        curr = curr->next;
    }

    printf("\nTotal price: %lld $\n", total);
    printf("Confirm purchase? (y/n): ");

    char buf[10];
    read_str(buf, 5);

    if (buf[0] == 'y' || buf[0] == 'Y')
    {
        curr = cart_head;
        while (curr != NULL)
        {
            curr->item->purchase_count += curr->quantity;
            curr = curr->next;
        }

        clear_cart();

        puts("Transaction successful");
    }
    else
    {
        puts("Transaction cancelled");
    }
}

void best_seller()
{
    int max_val = 0;
    int n = 10;

    for (int i = 0; i < n; i++)
    {
        if (products[i] && products[i]->purchase_count > max_val)
        {
            max_val = products[i]->purchase_count;
        }
    }

    int size_needed = (max_val + 1) * sizeof(int);
    int *count = (int *)malloc(size_needed);

    if (!count)
    {
        puts("Malloc failed");
        return;
    }

    memset(count, 0, size_needed);

    product *output[10] = {0}; 

    for (int i = 0; i < n; i++)
    {
        if (products[i])
        {
            count[products[i]->purchase_count]++;
        }
    }

    for (int i = 1; i <= max_val; i++)
    {
        count[i] += count[i - 1];
    }

    for (int i = n - 1; i >= 0; i--)
    {
        if (products[i])
        {
            int val = products[i]->purchase_count;
            int pos = count[val] - 1;
            output[pos] = products[i];
            count[val]--; 
        }
    }

    puts("\n--- Best Sellers ---");
    printf("%-20s | %-10s | %s\n", "Name", "Price", "Sold");
    puts("----------------------------------------------");

    for (int i = n - 1; i >= 0; i--)
    {
        if (output[i]) 
        {
            printf("%-20s | %-10lld | %d\n", 
                output[i]->name, 
                output[i]->price, 
                output[i]->purchase_count);
        }
    }
    puts("----------------------------------------------");

    free(count);
}
